‚ö° JavaScript Basics Crash Course

Why JavaScript for this role: Electron uses JavaScript/Node.js for GUI development. It's the foundation for modern cross-platform desktop apps. Coming from Python, you'll find JavaScript surprisingly similar in many ways.

1. Basic Syntax & Variables

Variable Declaration

// Three ways to declare variables

// let - block-scoped, can be reassigned
let age = 25;
age = 26;  // OK

// const - block-scoped, cannot be reassigned
const name = "Jeter";
// name = "John";  // ERROR!

// var - function-scoped (old way, avoid in modern code)
var oldStyle = "avoid this";

// No declaration (bad practice - creates global)
x = 5;  // DON'T DO THIS!
Python Comparison: JavaScript has let/const similar to Python's variable assignment, but with explicit scoping rules. Use const by default, let when you need to reassign.

3. Operators

// Arithmetic

let diff = 10 - 5;
let product = 10 * 5;

4. Control Flow

If-Else

let score = 85;

if (score >= 90) {
    console.log("Grade: A");
} else if (score >= 80) {
    console.log("Grade: B");
} else if (score >= 70) {
    console.log("Grade: C");
} else {
    console.log("Grade: F");
}

// Ternary operator
let result = score >= 60 ? "Pass" : "Fail";

// Short-circuit evaluation
let username = user && user.name;  // If user exists, get name
let defaultName = username || "Guest";  // Use default if username is falsy

Switch Statement

switch (day) {
    case "Monday":
        console.log("Start of week");
        break;  // IMPORTANT: prevents fall-through
    case "Friday":
        console.log("Almost weekend");
        break;
    case "Saturday":
    case "Sunday":
        console.log("Weekend!");
        break;
    default:
        console.log("Midweek");
}

5. Loops

For Loop

// Traditional for loop
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// For...of (iterate over values - like Python for)
let fruits = ["apple", "banana", "orange"];
for (let fruit of fruits) {
    console.log(fruit);
}

// For...in (iterate over keys/indices)
let person = { name: "Jeter", age: 25 };
for (let key in person) {
    console.log(`${key}: ${person[key]}`);
}

While Loop

let count = 0;
while (count < 5) {
    console.log(count);
    count++;
}

// Do-while (executes at least once)
let num = 0;
do {
    console.log(num);
    num++;
} while (num < 5);

Loop Control

// Break - exit loop
for (let i = 0; i < 10; i++) {
    if (i === 5) break;
    console.log(i);
}

// Continue - skip to next iteration
for (let i = 0; i < 10; i++) {
    if (i % 2 === 0) continue;
    console.log(i);  // Only odd numbers
}

7. Arrays (like Python lists)

Array Methods

let numbers = [1, 2, 3, 4, 5];

// Add/Remove
numbers.push(6);           // Add to end: [1,2,3,4,5,6]
numbers.pop();             // Remove from end: [1,2,3,4,5]
numbers.unshift(0);        // Add to start: [0,1,2,3,4,5]
numbers.shift();           // Remove from start: [1,2,3,4,5]

// Access
let first = numbers[0];
let last = numbers[numbers.length - 1];

// Slice (like Python slicing)
let subset = numbers.slice(1, 3);  // [2, 3] (doesn't modify original)

// Splice (remove/insert elements)
numbers.splice(2, 1);      // Remove 1 element at index 2
numbers.splice(2, 0, 99);  // Insert 99 at index 2

// Join (like Python join)
let text = numbers.join(", ");  // "1, 2, 99, 4, 5"

// Common operations
let length = numbers.length;
let includes = numbers.includes(3);  // true
let index = numbers.indexOf(99);     // 2

Array Iteration Methods

let numbers = [1, 2, 3, 4, 5];

// forEach (execute function for each element)
numbers.forEach(num => {
    console.log(num);
});

// map (transform each element - like Python map())
let doubled = numbers.map(n => n * 2);
// [2, 4, 6, 8, 10]

// filter (select elements - like Python filter())
let evens = numbers.filter(n => n % 2 === 0);
// [2, 4]

// reduce (accumulate - like Python reduce())
let sum = numbers.reduce((acc, n) => acc + n, 0);
// 15

// find (find first matching element)
let found = numbers.find(n => n > 3);
// 4

// some (check if any element matches)
let hasEven = numbers.some(n => n % 2 === 0);
// true

// every (check if all elements match)
let allPositive = numbers.every(n => n > 0);
// true

// Complex example (chaining)
let result = numbers
    .filter(n => n > 2)
    .map(n => n * 2)
    .reduce((acc, n) => acc + n, 0);
// (3,4,5) -> (6,8,10) -> 24

Object Creation and Access

Object Methods

let person = {
    name: "Jeter",
    age: 25,
    
    // Method (function inside object)
    greet: function() {
        console.log(`Hello, I'm ${this.name}`);
    },
    
    // Shorthand method syntax
    celebrate() {
        console.log(`${this.name} is celebrating!`);
    }
};

person.greet();      // "Hello, I'm Jeter"
person.celebrate();  // "Jeter is celebrating!"

Object Destructuring

let person = { name: "Jeter", age: 25, city: "New York" };

// Extract properties
let { name, age } = person;
console.log(name);  // "Jeter"
console.log(age);   // 25

// With different variable names
let { name: fullName, age: years } = person;
console.log(fullName);  // "Jeter"

// With default values
let { name, country = "USA" } = person;
console.log(country);  // "USA"

Spread Operator

// Copy object
let original = { a: 1, b: 2 };
let copy = { ...original };

// Merge objects
let person = { name: "Jeter" };
let details = { age: 25, city: "New York" };
let complete = { ...person, ...details };
// { name: "Jeter", age: 25, city: "New York" }

// Copy array
let arr1 = [1, 2, 3];
let arr2 = [...arr1];

// Merge arrays
let combined = [...arr1, ...arr2];

9. Classes (ES6+)

class Person {
    // Constructor
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    
    // Method
    greet() {
        console.log(`Hi, I'm ${this.name}`);
    }
    
    // Getter
    get info() {
        return `${this.name}, ${this.age}`;
    }
    
    // Setter
    set updateAge(newAge) {
        this.age = newAge;
    }
    
    // Static method (called on class, not instance)
    static create(name, age) {
        return new Person(name, age);
    }
}

// Usage
let person = new Person("Jeter", 25);
person.greet();  // "Hi, I'm Jeter"

// Inheritance
class Employee extends Person {
    constructor(name, age, jobTitle) {
        super(name, age);  // Call parent constructor
        this.jobTitle = jobTitle;
    }
    
    work() {
        console.log(`${this.name} is working as ${this.jobTitle}`);
    }
}

let emp = new Employee("Jeter", 25, "Software Engineer");
emp.greet();  // Inherited method
emp.work();   // Own method

10. Async/Await (Critical for GUI)

Promises

// Promise represents eventual completion of async operation
const fetchData = () => {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve("Data loaded");
        }, 2000);
    });
};

// Using .then()
fetchData()
    .then(data => console.log(data))
    .catch(error => console.error(error));

Async/Await (Modern Way)

// Async function
async function loadData() {
    try {
        let data = await fetchData();
        console.log(data);
    } catch (error) {
        console.error(error);
    }
}

// Real-world example: API call
async function getUserData(userId) {
    try {
        const response = await fetch(`https://api.example.com/users/${userId}`);
        const data = await response.json();
        return data;
    } catch (error) {
        console.error("Error fetching user:", error);
    }
}

// GUI example: Button click with async operation
async function handleButtonClick() {
    // Show loading
    document.getElementById('status').textContent = 'Loading...';
    
    // Fetch data (doesn't block UI)
    const data = await getUserData(123);
    
    // Update UI
    document.getElementById('result').textContent = data.name;
    document.getElementById('status').textContent = 'Ready';
}

Template Literals

    Hello, ${name}!
    You are ${age} years old.
    Next year you'll be ${age + 1}.
`;

// Expression evaluation

Optional Chaining

// Safe property access
const user = {
    profile: {
        name: "Jeter"
    }
};

// Without optional chaining (can error)
// const city = user.profile.address.city;  // ERROR if address doesn't exist

// With optional chaining (returns undefined if doesn't exist)
const city = user.profile?.address?.city;  // undefined, no error

Nullish Coalescing

// Use ?? to provide default for null/undefined only
let username = null;
let display = username ?? "Guest";  // "Guest"

// Different from || which treats "" and 0 as falsy
let count = 0;
let result1 = count || 10;   // 10 (0 is falsy)
let result2 = count ?? 10;   // 0 (0 is not null/undefined)
